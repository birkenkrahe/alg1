     #include <iostream>

     int main()
     {
       int time;
       cout << "Enter time in seconds:\n";
       cin >> time;
       cout << "You entered: " << time << " seconds." << endl;
       int answer = (32 * time * time) / 2;
       cout << "The distance is ";
       cout << answer;

   #+end_src

   squared (or 9.8 meter per second squared in the metric system) is
